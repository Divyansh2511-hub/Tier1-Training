#include <stdio.h>

int main()
{
    int arr[100],n;
    int i, max1 , max2;
    
    printf("Enter size of array: ");
    scanf("%d",&n);
    
    printf("Enter number of elements :\n");
    for(i=0;i<=n;i++){
        scanf("%d", &arr[i]);
    }
    max1 = max2 = 0;
    
    for(i=0;i<n;i++){
        if (arr[i]>max1){
            max2 = max1;
            max1 = arr[i];
        } 
        else if (arr[i]>max2 && arr[i] != max1 ){
            max2 = arr[i];
        }
    }
    printf("Second Largest  is %d\n",max2);
    return 0;
}

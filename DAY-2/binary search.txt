#include <stdio.h>

int main() {
    int n, i , key ;
    
    printf("Enter Array Size: ");
    scanf("%d",&n);
    int arr[n];
    printf("Enter elements in Sorted form : \n");
    for (i=0;i<n;i++) {
        scanf("%d", &arr[i]);
    }
    
    printf("Enter Number to find:");
    scanf("%d", &key);
    
    int low = 0 , high = n-1 , mid;
    int found = 0;
    
    while (low<=high)
    {
        if (arr[mid]==key)
        {
            printf("Found at %d\n", mid);
            found = 1;
            break;
        }
        else if (key> arr[mid])
        {
        low = mid+1;
        }
        else
        {
        high = mid - 1;
        }
    }
    
    if (!found)
    {
        printf("element not found");
    }
    return 0;
}

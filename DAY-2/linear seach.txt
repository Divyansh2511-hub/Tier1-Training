#include <stdio.h>

int main() {
    int n, i , key ;
    int found=0;
    
    printf("Enter Array Size: ");
    scanf("%d",&n);
    int arr[n];
    printf("Enter elements : \n",n);
    for (i=0;i<n;i++) {
        scanf("%d", &arr[i]);
    }
    
    printf("Enter Number to find:");
    scanf("%d", &key);
    
    for(i=0;i<n;i++) {
        if(arr[i]==key) {
            printf("Element found at %d \n", i);
            found = 1;
            break;
        }
    }
    if (!found){
        printf("element not found");
    }
    return 0;
}

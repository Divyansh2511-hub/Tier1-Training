#include <stdio.h>

int main() {
    int num1 , num2 ;
    int freq1[10] = {0};
    int freq2[10] = {0};
    int digit;
    
    printf("Enter your first number\n");
    scanf("%d", &num1);
    
    printf("Enter your second number\n");
    scanf("%d", &num2);
    
    int temp = num1 ;
    while(temp>0){
        digit = temp%10;
        freq1[digit]++;
        temp /= 10;
    }
    
    temp = num2 ;
    while(temp>0){
        digit = temp%10;
        freq2[digit]++;
        temp /=10;
    }
    
    int i , isAnagram = 1;
    for (i = 0; i < 10; i++) {
        if (freq1[i] != freq2[i]) {
            isAnagram = 0;
            break;
        }
    }
    
    if (isAnagram){
        printf ("Number is anagram");
    }
    else{
        printf("Number is not anagram");
    }
    return 0;
}
